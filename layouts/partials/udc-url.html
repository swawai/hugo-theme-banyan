{{/* 构建 UDC term 的 canonical 链接，并按需附带 sort query。 */}} {{- $ctx := .ctx -}} {{- $mode := .mode | default "bydate" -}} {{- $plural := printf "%v" (.plural | default "") -}} {{- if eq $plural "" -}}{{- $plural = printf "%v" ($ctx.Data.Plural | default "udc") -}}{{- end -}} {{- $href := printf "%s%s/" (printf "/%s/" $plural | relLangURL) (urlize .value) -}} {{- return (partial "sort-url.html" (dict "href" $href "mode" $mode)) -}}
