{{- $raw := printf "%v" . -}} {{- $result := $raw -}} {{- if and $raw (in $raw "?") -}} {{- $parts := split $raw "#" -}} {{- $base := index $parts 0 -}} {{- $hash := "" -}} {{- if gt (len $parts) 1 -}} {{- $hash = printf "#%s" (delimit (after 1 $parts) "#") -}} {{- end -}} {{- $baseParts := split $base "?" -}} {{- $path := index $baseParts 0 -}} {{- $query := delimit (after 1 $baseParts) "?" -}} {{- $filtered := slice -}} {{- range (split $query "&") -}} {{- $part := printf "%v" . -}} {{- if $part -}} {{- $pair := split $part "=" -}} {{- $key := lower (index $pair 0) -}} {{- if ne $key "sort" -}} {{- $filtered = $filtered | append $part -}} {{- end -}} {{- end -}} {{- end -}} {{- $result = $path -}} {{- if gt (len $filtered) 0 -}} {{- $result = printf "%s?%s" $path (delimit $filtered "&") -}} {{- end -}} {{- if $hash -}} {{- $result = printf "%s%s" $result $hash -}} {{- end -}} {{- end -}} {{- return $result -}}
