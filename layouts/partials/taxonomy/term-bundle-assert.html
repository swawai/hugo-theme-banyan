\r\n\t" -}} {{- if and (ne $plural "") (ne $value "") -}} {{- $pagePath := "" -}} {{- with $page.File -}} {{- $pagePath = .Path -}} {{- else -}} {{- $pagePath = $page.RelPermalink | default $page.Title | default "(unknown page)" -}} {{- end -}} {{- $lang := $page.Site.Language.Lang | default "default" -}} {{- $defaultLang := "" -}} {{- with index $page.Site.Sites 0 -}} {{- $defaultLang = .Language.Lang | default "" -}} {{- end -}} {{- $unassignedTerm := printf "%v" (index $spec "unassigned_term" | default "") -}} {{- if not (and (ne $unassignedTerm "") (eq (partial "taxonomy/normalize.html" (dict "spec" $spec "value" $value)) (partial "taxonomy/normalize.html" (dict "spec" $spec "value" $unassignedTerm)))) -}} {{- $targets := slice $value -}} {{- if eq (printf "%v" (index $spec "mode")) "tree" -}} {{- $separator := printf "%v" (index $spec "separator" | default "/") -}} {{- $segments := split $value $separator -}} {{- $path := slice -}} {{- $targets = slice -}} {{- range $seg := $segments -}} {{- if eq $seg "" -}}{{- continue -}}{{- end -}} {{- $path = $path | append $seg -}} {{- $targets = $targets | append (delimit $path $separator) -}} {{- end -}} {{- end -}} {{- $seen := dict -}} {{- range $target := $targets -}} {{- $targetText := trim (printf "%v" $target) " \r\n\t" -}} {{- if eq $targetText "" -}}{{- continue -}}{{- end -}} {{- $norm := partial "taxonomy/normalize.html" (dict "spec" $spec "value" $targetText) -}} {{- if isset $seen $norm -}}{{- continue -}}{{- end -}} {{- $seen = merge $seen (dict $norm true) -}} {{- $termPage := partialCached "taxonomy/term-page-get.html" (dict "ctx" $page "spec" $spec "plural" $plural "value" $targetText) $lang (lower $plural) $norm -}} {{- $expected := printf "content/%s/%s/_index.%s.md" $plural $targetText $lang -}} {{- $expectedHint := $expected -}} {{- if eq $lang $defaultLang -}} {{- $expectedHint = printf "%s（默认语言也可写为 content/%s/%s/_index.md）" $expected $plural $targetText -}} {{- end -}} {{- if eq $termPage "" -}} {{- errorf "Build failed: %s uses taxonomy %q term %q, but no matching term bundle exists. Create %s first." $pagePath $plural $targetText $expectedHint -}} {{- end -}} {{- $hasFile := false -}} {{- with $termPage.File -}} {{- $hasFile = true -}} {{- end -}} {{- if not $hasFile -}} {{- errorf "Build failed: %s uses taxonomy %q term %q, but language %q has no explicit term bundle. Create %s first; do not rely on autogenerated pages." $pagePath $plural $targetText $lang $expectedHint -}} {{- end -}} {{- $title := trim (printf "%v" ($termPage.LinkTitle | default $termPage.Title | default "")) " \r\n\t" -}} {{- if eq $title "" -}} {{- $bundlePath := $termPage.File.Path | default $expected -}} {{- errorf "Build failed: %s uses taxonomy %q term %q, but %s is missing a non-empty title." $pagePath $plural $targetText $bundlePath -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
