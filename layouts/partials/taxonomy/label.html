{{- $ctx := .ctx | default . -}} {{- $spec := .spec | default dict -}} {{- $plural := printf "%v" (index $spec "plural") -}} {{- $page := partial "taxonomy/page-get.html" (dict "ctx" $ctx "plural" $plural) -}} {{- $pagePath := $page.File.Path | default (printf "content/%s/_index.*" $plural) -}} {{- $title := trim (printf "%v" ($page.Title | default "")) " \r\n\t" -}} {{- if eq $title "" -}} {{- errorf "编译失败: %s 必须提供非空 title，作为 taxonomy %q 的唯一命名来源。" $pagePath $plural -}} {{- end -}} {{- return $title -}}
