{{- $ctx := .ctx | default . -}} {{- $spec := .spec | default dict -}} {{- $plural := printf "%v" (index $spec "plural") -}} {{- $page := partial "taxonomy/page-get.html" (dict "ctx" $ctx "plural" $plural) -}} {{- $pagePath := $page.File.Path | default (printf "content/%s/_index.*" $plural) -}} {{- $title := trim (printf "%v" ($page.Title | default "")) " \r\n\t" -}} {{- if eq $title "" -}} {{- errorf "Build failed: %s must provide a non-empty title as the single naming source for taxonomy %q." $pagePath $plural -}} {{- end -}} {{- return $title -}}
