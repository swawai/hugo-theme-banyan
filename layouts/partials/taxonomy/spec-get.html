{{- $ctx := .ctx | default . -}} {{- $plural := lower (printf "%v" .plural) -}} {{- $page := partial "taxonomy/page-get.html" (dict "ctx" $ctx "plural" $plural) -}} {{- $pagePath := $page.File.Path | default (printf "content/%s/_index.*" $plural) -}} {{- $pageTitle := trim (printf "%v" ($page.Title | default "")) " \r\n\t" -}} {{- if eq $pageTitle "" -}} {{- errorf "Build failed: %s must provide a non-empty title as the single naming source for taxonomy %q." $pagePath $plural -}} {{- end -}} {{- if not (isset $page.Params "banyan_taxonomy") -}} {{- errorf "Build failed: %s is missing [banyan_taxonomy]. Rendering config for taxonomy %q may only be defined in content/%s/_index..md." $pagePath $plural $plural -}} {{- end -}} {{- $taxonomyMeta := index $page.Params "banyan_taxonomy" -}} {{- $forbiddenKeys := slice "label" "home_label" -}} {{- range $key := $forbiddenKeys -}} {{- if isset $taxonomyMeta $key -}} {{- errorf "Build failed: %s uses removed key [banyan_taxonomy].%s. All display naming for taxonomy %q must now come from the title of the same bundle." $pagePath $key $plural -}} {{- end -}} {{- end -}} {{- $requiredKeys := slice "mode" "show_in_home" "home_weight" "article_weight" "normalize" "article_mode" -}} {{- range $key := $requiredKeys -}} {{- if not (isset $taxonomyMeta $key) -}} {{- errorf "Build failed: %s is missing [banyan_taxonomy].%s. Taxonomy %q no longer provides defaults or fallback sources." $pagePath $key $plural -}} {{- end -}} {{- end -}} {{- $spec := dict "plural" $plural "separator" "/" "require_term_bundles" true "term_rel" "" "unassigned_term" "" "unassigned_label" "" -}} {{- $spec = merge $spec $taxonomyMeta (dict "plural" $plural) -}} {{- $allowedModes := slice "flat" "tree" -}} {{- $mode := lower (printf "%v" (index $spec "mode")) -}} {{- if not (in $allowedModes $mode) -}} {{- errorf "Build failed: %s only supports flat/tree for [banyan_taxonomy].mode; got %q." $pagePath (index $spec "mode") -}} {{- end -}} {{- $allowedNormalize := slice "identity" "lower" -}} {{- $normalize := lower (printf "%v" (index $spec "normalize")) -}} {{- if not (in $allowedNormalize $normalize) -}} {{- errorf "Build failed: %s only supports identity/lower for [banyan_taxonomy].normalize; got %q." $pagePath (index $spec "normalize") -}} {{- end -}} {{- $allowedArticleModes := slice "all" "deepest_by_root" "none" -}} {{- $articleMode := lower (printf "%v" (index $spec "article_mode")) -}} {{- if not (in $allowedArticleModes $articleMode) -}} {{- errorf "Build failed: %s only supports all/deepest_by_root/none for [banyan_taxonomy].article_mode; got %q." $pagePath (index $spec "article_mode") -}} {{- end -}} {{- $spec = merge $spec (dict "mode" $mode "normalize" $normalize "article_mode" $articleMode "require_term_bundles" (eq (lower (printf "%v" (index $spec "require_term_bundles"))) "true") ) -}} {{- return $spec -}}
