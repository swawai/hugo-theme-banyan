{{/* 收集给定 prefix 的 TAGS 子树页面（参考 UDC 版本）。 参数： ctx: 页面上下文 (.) prefix: 要收集的 term 前缀（如 "Hugo"、"Hugo/fugo" 等） mode: subtree|exact （默认 subtree） 返回：页面切片 */}} {{ $page := .ctx }} {{ $prefix := .prefix }} {{ $mode := cond (isset . "mode") .mode "subtree" }} {{ $lang := $page.Site.Language.Lang }} {{ $key := printf "tags:%s:%s:%s" $mode $prefix $lang }} {{ $scr := $page.Scratch }} {{ $res := newScratch }} {{ $cached := $scr.Get $key }} {{ if $cached }} {{ $res.Set "v" $cached }} {{ else }} {{ $pagesAll := slice }} {{ if eq $mode "subtree" }} {{ range $t, $pages := site.Taxonomies.tags }} {{ if or (eq $t $prefix) (hasPrefix $t (printf "%s/" $prefix)) }} {{ $pagesAll = $pagesAll | append $pages }} {{ end }} {{ end }} {{ else if eq $mode "exact" }} {{ $exact := index site.Taxonomies.tags $prefix }} {{ with $exact }} {{ range . }} {{ $arr := .Params.tags }} {{ $hit := false }} {{ if $arr }} {{ range $arr }} {{ if eq (printf "%v" .) $prefix }}{{ $hit = true }}{{ end }} {{ end }} {{ end }} {{ if $hit }} {{ $pagesAll = $pagesAll | append . }} {{ end }} {{ end }} {{ end }} {{ end }} {{/* 去重并缓存 */}} {{ $seen := dict }} {{ $uniq := slice }} {{ range $pagesAll }} {{ $k := .RelPermalink }} {{ if not (isset $seen $k) }} {{ $seen = merge $seen (dict $k true) }} {{ $uniq = $uniq | append . }} {{ end }} {{ end }} {{ $scr.Set $key $uniq }} {{ $res.Set "v" $uniq }} {{ end }} {{ return ($res.Get "v") }}
