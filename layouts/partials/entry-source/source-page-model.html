{{- $page := .page -}} {{- $logicalPath := printf "%v" (.logical_path | default "") -}} {{- $provider := printf "%v" (.provider | default "") -}} {{- if not $page -}} {{- errorf "Build failed: entry-source/source-page-model.html requires page." -}} {{- end -}} {{- if eq $logicalPath "" -}} {{- errorf "Build failed: entry source page %q is missing logical_path." ($page.File.Path | default (printf "%s %s" $page.Kind $page.RelPermalink)) -}} {{- end -}} {{- partial "slots/prepare.html" (dict "ctx" $page) -}} {{- $canonicalModel := partial "breadcrumb/canonical-model.html" (dict "ctx" $page "slot" "breadcrumb") -}} {{- $rootRaw := index $canonicalModel "root_item" | default dict -}} {{- $rootText := trim (printf "%v" (index $rootRaw "text")) " " -}} {{- $rootHref := trim (printf "%v" (index $rootRaw "href")) " " -}} {{- if or (eq $rootText "") (eq $rootHref "") -}} {{- errorf "Build failed: entry source page %q has a root item missing text or href." ($page.File.Path | default (printf "%s %s" $page.Kind $page.RelPermalink)) -}} {{- end -}} {{- $rootItem := dict "text" $rootText "href" $rootHref -}} {{- with index $rootRaw "title" }}{{ $rootItem = merge $rootItem (dict "title" (printf "%v" .)) }}{{ end -}} {{- $rootMenuItems := index $canonicalModel "root_menu" | default (slice $rootItem) -}} {{- $rootMenuLabel := trim (printf "%v" (index $canonicalModel "root_menu_label" | default "Breadcrumb root")) " " -}} {{- $tailItems := index $canonicalModel "tail_items" | default slice -}} {{- $canonicalLevels := index $canonicalModel "levels" | default slice -}} {{- $levels := slice -}} {{- range $level := $canonicalLevels -}} {{- $item := index $level "item" | default dict -}} {{- $collectionLogicalPath := partial "entry-source/logical-path-from-href.html" (dict "ctx" $page "provider" $provider "href" (index $level "collection_href") ) -}} {{- $collectionSource := partial "entry-source/collection-source.html" (dict "provider" $provider "logical_path" $collectionLogicalPath ) -}} {{- $levelModel := dict "item" $item -}} {{- if gt (len $collectionSource) 0 -}} {{- $levelModel = merge $levelModel (dict "collection_source" $collectionSource) -}} {{- end -}} {{- $levels = $levels | append $levelModel -}} {{- end -}} {{- $currentCollectionSource := partial "entry-source/collection-source.html" (dict "provider" $provider "logical_path" $logicalPath ) -}} {{- $currentCollectionItems := dict -}} {{- $entryHrefs := slice -}} {{- if in (slice "all" "products" "section-d" "taxonomy") $provider -}} {{- $currentCollectionItemsJson := partial (printf "entry-source/items-json-%s.html" $provider) (dict "source" (dict "page" $page "provider" $provider "logical_path" $logicalPath ) ) -}} {{- if ne (trim $currentCollectionItemsJson " \r\n\t") "" -}} {{- $currentCollectionItems = transform.Unmarshal (dict "format" "json") $currentCollectionItemsJson -}} {{- $entryHrefs = partial "entry-source/items-payload-page-hrefs.html" (dict "payload" $currentCollectionItems) | default slice -}} {{- end -}} {{- end -}} {{- $result := dict "provider" $provider "page" $page "logical_path" $logicalPath "root_item" $rootItem "root_menu" $rootMenuItems "root_menu_label" $rootMenuLabel "tail_items" $tailItems "levels" $levels "current_collection_source" $currentCollectionSource "current_collection_items" $currentCollectionItems "entry_hrefs" $entryHrefs -}} {{- return $result -}}
