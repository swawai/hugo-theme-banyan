{{- $ctx := .ctx -}} {{- if not $ctx -}} {{- errorf "编译失败: entry-source/page-entry-key.html 缺少 ctx。" -}} {{- end -}} {{- $result := "" -}} {{- with $ctx.RelPermalink -}} {{- $trimmedPath := replaceRE "/+$" "" . -}} {{- if ne $trimmedPath "" -}} {{- $result = urlize (path.Base $trimmedPath) -}} {{- end -}} {{- end -}} {{- if eq $result "" -}} {{- $fallback := trim (printf "%v" ($ctx.Params.slug | default "")) " " -}} {{- if eq $fallback "" -}} {{- $fallback = trim (printf "%v" ($ctx.Title | default "")) " " -}} {{- end -}} {{- $result = urlize $fallback -}} {{- end -}} {{- if or (eq $result "") (eq $result ".") (eq $result "..") -}} {{- errorf "编译失败: 页面 %q 无法生成稳定的 entry key。" ($ctx.File.Path | default (printf "%s %s" $ctx.Kind $ctx.RelPermalink)) -}} {{- end -}} {{- return $result -}}
