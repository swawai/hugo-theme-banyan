{{- $page := .page -}} {{- if not $page -}} {{- errorf "Build failed: products/model.html requires page." -}} {{- end -}} {{- $pageRef := $page.File.Path | default (printf "%s %s" $page.Kind $page.RelPermalink) -}} {{- $product := index $page.Params "product" -}} {{- if not $product -}} {{- errorf "Build failed: page %q is missing the product object." $pageRef -}} {{- end -}} {{- if eq (trim (printf "%v" $product) " ") "true" -}} {{- errorf "Build failed: page %q must use an object for product; the legacy product=true format is no longer supported." $pageRef -}} {{- end -}} {{- $origin := lower (trim (printf "%v" (index $product "origin" | default "")) " ") -}} {{- if not (in (slice "first-party" "third-party") $origin) -}} {{- errorf "Build failed: page %q has unsupported product.origin=%q; expected first-party or third-party." $pageRef $origin -}} {{- end -}} {{- $price := index $product "price" -}} {{- if not $price -}} {{- errorf "Build failed: page %q is missing product.price." $pageRef -}} {{- end -}} {{- $amountRaw := index $price "amount" -}} {{- $amountText := trim (printf "%v" $amountRaw) " " -}} {{- if eq $amountText "" -}} {{- errorf "Build failed: page %q is missing product.price.amount." $pageRef -}} {{- end -}} {{- if eq (len (findRE `^[0-9]+(?:\.[0-9]+)?$` $amountText)) 0 -}} {{- errorf "Build failed: page %q has invalid product.price.amount=%q." $pageRef $amountText -}} {{- end -}} {{- $currency := upper (trim (printf "%v" (index $price "currency" | default "")) " ") -}} {{- if eq $currency "" -}} {{- errorf "Build failed: page %q is missing product.price.currency." $pageRef -}} {{- end -}} {{- $valueText := trim (printf "%v" (index $product "value" | default "")) " " -}} {{- if eq $valueText "" -}} {{- errorf "Build failed: page %q is missing product.value." $pageRef -}} {{- end -}} {{- $isFree := gt (len (findRE `^0(?:\.0+)?$` $amountText)) 0 -}} {{- $scope := "paid" -}} {{- if eq $origin "third-party" -}} {{- $scope = "third-party" -}} {{- else if $isFree -}} {{- $scope = "free" -}} {{- end -}} {{- $priceText := printf "%s %s" $currency $amountText -}} {{- if $isFree -}} {{- $priceText = i18n "products_free" -}} {{- end -}} {{- return (dict "origin" $origin "scope" $scope "price_amount_text" $amountText "price_currency" $currency "price_text" $priceText "value_text" $valueText ) -}}
